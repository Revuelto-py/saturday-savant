{% set glossary = [ ('SP+', 'Bill Connelly\'s tempo- and opponent-adjusted efficiency metric. Combines offense, defense, and special teams into a single predictive rating. Higher is better.'), ('EPA / Play', 'Expected Points Added per play. Measures how much a team moves the needle toward scoring relative to what\'s expected given down, distance, and field position. The single best measure of efficiency.'), ('Success Rate', 'The percentage of plays deemed "successful" — gaining 50% of yards needed on 1st down, 70% on 2nd, or 100% on 3rd/4th. Measures consistency.'), ('Explosiveness', 'Average EPA on successful plays only. Measures the ability to create big gains when things go right. High explosiveness with high success rate = elite offense.'), ('Power Success', 'Percentage of short-yardage situations (2nd/3rd/4th and ≤2 yards) converted. Measures ability to win in the trenches.'), ('Stuff Rate', 'Percentage of rush attempts stopped at or behind the line of scrimmage. High offensive stuff rate = bad O-line. High defensive stuff rate = dominant D-line.'), ('Line Yards / Carry', 'Rushing yards credited to the offensive line (first 5 yards). Isolates O-line performance from the running back\'s contribution.'), ('2nd Level Yards', 'Rushing yards gained 5–10 yards past the line of scrimmage. Represents yards where linebackers have primary responsibility.'), ('Open Field Yards', 'Rushing yards gained beyond 10 yards. Largely a function of the running back\'s speed and elusiveness in space.'), ('Rush / Pass EPA', 'EPA per play broken down by run vs pass. Helps identify whether a team\'s success comes from the ground or through the air.'), ('Rush / Pass Success Rate', 'Success rate broken down by run vs pass. Shows whether a team is consistently effective in each phase.'), ('Rush / Pass Explosiveness', 'Average EPA on successful rush or pass plays only. Identifies big-play threats in each phase of the offense.'), ('Havoc Rate', 'The percentage of opponent plays this defense disrupts — tackles for loss, forced fumbles, pass breakups, and interceptions. Broken down into Front 7 (line/linebacker) and DB (secondary) contributions. Higher is better.'), ('Field Position', 'Average yards a drive needs to travel to reach the end zone, based on where it starts. For "Avg Starting Field Position," lower is better — the offense starts closer to the opponent\'s end zone. For "Opponent Starting Field Position," higher is better — the defense (and special teams) is pinning opponents back, forcing them to go further.'), ] %}
